    function Durchschnittsnote($schueler) {
        $summe = 0;
        $anzahl = count($schueler);
        
        foreach ($schueler as $student) {
            $summe += $student["note"];
        }
        if ($anzahl > 0) {
            return $summe / $anzahl;
        } else {
            return 0;
        }
    }
    $durchschnittsnote = Durchschnittsnote($schueler);
    echo "Durchschnittsnote: " . $durchschnittsnote;

    <h2>Übung 4: Prime-Zahlen finden</h2>
    <?php
    function isPrime($n) {
        if ($n <= 1) {
            return false; 
        }
        for ($i = 2; $i < $n; $i++) {
            if ($n % $i == 0) {
                return false; 
            }
        }
        return true;
    }
    for ($n = 1; $n <= 100; $n++) {
        if (isPrime($n)) {
            echo $n . "<br>";
        }
    }
    ?>

        ["name" => "Banane", "preis" => 0.3, "menge" => 20],
        ["name" => "Orange", "preis" => 0.8, "menge" => 15],

    echo "</tr>";

    foreach ($produkte as $supermarkt) {
        $gesamtpreis = $supermarkt['preis'] * $supermarkt['menge'];
        echo "<tr>";
        echo "<td>" . $supermarkt['name'] . "</td>";
        echo "<td>" . $supermarkt['preis'] . "</td>";
        echo "<td>" . $supermarkt['menge'] . "</td>";
        echo "<td>$gesamtpreis</td>";
        echo "</tr>";
    }
    echo "</table>";
    echo "<br>";

    function Gesamtwert($produkte) {
        $summe = 0;
        foreach ($produkte as $supermarkt) {
            $gesamtpreis = $supermarkt['preis'] * $supermarkt['menge'];
            $summe += $gesamtpreis; 
        }
        return $summe; 
    }
    $gesamtwert = Gesamtwert($produkte);
    echo "Der Gesamtwert des Warenbestandes beträgt " . $gesamtwert . " €.";
    ?>
